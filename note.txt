def clean_html(raw):
    cleaner = re.compile('<.*?>')
    cleaned_text = re.sub(cleaner, '', raw)
    
    return cleaned_text

def remove_tags(text):
    parse = etree.HTMLParser(recover=True)
    
    return ''.join(ElementTree.fromstring(text, parser=parse).itertext())
